Motion pages are downloaded to memory 0x8000000 + 0x43200
Motion pages strcuture:

typedef struct {
u16 joint_pos[26];
u16 padding[5];
u8 pause; // multiply by 8 to get pause time in ms;
u8 time; // 1 unit = 8ms
} motion_step;

typedef struct {
char name[15]; // including null at the end
u8 repetitions;
u8 uk1[4];
u8 n_steps;
u8 uk1a;
u8 speed_rate; //multiply pause and play time, in ms, for each step, with Speed rate, then divide by 0x20 = 32 to get real time in ms.
u8 uk2;
u8 ctrl_inertial_force;
u8 next_page;
u8 exit_page;
u8 uk3[5];
u8 joint_softness[26]; // aka Compliance slope
u8 uk4[6];
motion_step steps[7];
} motion_page;


(speedrate10 is the old motion protocol value which is just "speed rate" from r+ motion * 10)
(speed_rate is the actual value stored in memory by r+ motion)
(t is time (pause/step) in ms)
t * 10 / speedrate10 = t * speed_rate / 32
speedrate10 = 32 * 10 / speed_rate
speed_rate = 32 * 10 / speedrate10 

+0	Name 15 bytes incl ending 0 [15 bytes]
+F	Repeat time [1 byte]
+10	- [always 0a ?]
+11	- [always 0 ?] [3 bytes]
+14	Number of steps [1 byte] (value range 0-7: max 7 steps)
+15	?
+16	Speed rate (multiply pause and play time, in ms, for each step, with Speed rate, then divide by 0x20 = 32 to get real time in ms) [1 byte]
+17	?
+18	Ctrl intertial force: 0-127 [1 byte]
+19	next page [1 byte] 
+1A	exit page [1 byte] 
+1B	?
+1C	?
+1D	?
+1E	?
+1F	Motion checksum? [1 byte]
+20	idx 0 Compliance (joint softness) cc, ccw [1 pair in 1 byte, e.g. value 0x55 => 5 cw and 5 ccw]
+21	idx 1 Compliance cc, ccw (motor with id 1) [1 bytes]
+22	idx 2 Compliance cc, ccw (motor with id 2) [1 bytes]
...... (26 value pairs total, idx 1 to 18 is for the motors, others spare?)

+2D	0
